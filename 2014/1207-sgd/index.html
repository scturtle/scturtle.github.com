SGD

In [13]: 
def add_const(x, const=np.array([1.])):
    return np.r_[x, const]

def predict(w, x):
    wTx = np.dot(w, add_const(x))
    return 1. / (1. + exp(-max(min(wTx, 10.), -10.)))

def update(w, x, y, alpha):
    p = predict(w, x)
    w -= alpha * (p - y) * add_const(x)
    
def train(X, y, alpha, w=None):
    n, m = X.shape
    if w is None:
        w = np.zeros(m + 1)
    for i in xrange(n):
        update(w, X[i,:], y[i], alpha)
    return w
