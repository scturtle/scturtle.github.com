SGD

Original gradient descent update step:

$w \leftarrow w - \frac \alpha n \cdot (p(X)-y) \cdot X$

Stochastic gradient descent:

$w \leftarrow w - \alpha \cdot (p(X_i) - y_i) \cdot X_i$

def predict(w, x):
    wTx = np.dot(w, np.append(x, 1))
    return 1. / (1. + exp(- wTx ))

def update(w, x, y, alpha):
    p = predict(w, x)
    w -= alpha * (p - y) * np.append(x, 1)
    
def train(X, y, alpha, w=None):
    n, m = X.shape
    if w is None:
        w = np.zeros(m + 1)
    for i in xrange(n):
        update(w, X[i,:], y[i], alpha)
    return w
